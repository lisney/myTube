```
<script>
    const list = document.querySelector('ul')

    let data =['디자인', '데이터', '스터디', '간식']

    let jsBook = {
        title:'자바스크립트입문',
        price:25000,
        stock: 3
    }

    data.forEach(i=>{
        const li = document.createElement('li')
        li.innerHTML=`${i}`
        list.appendChild(li)
    })
    console.log(jsBook.title)
    jsBook.title='파이썬'
    console.log(jsBook.title)
</script>
```
